Modern cosmology supplement

Legendre

$$
\begin{aligned}
& P_0(\mu)=1 \
& P_1(\mu)=\mu \
& P_2(\mu)=\frac{3 \mu^2-1}{2} \
& P_3(\mu)=\frac{5}{2} \mu^3-\frac{3}{2} \mu \
\end{aligned}
$$

$$
\begin{aligned}
& \mu P_0(\mu)=P_1(m) \
& \begin{array}{l}
\mu P_1(\mu)=\mu^2=\frac{2}{3} P_2(\mu)+\frac{1}{3} P_0\left(\mu\right)
\end{array} \
& \mu P_2=\frac{3}{5}P_3+\frac{2}{5}P_1 \

&\mu P_\ell=\frac{(\ell+1)P_{\ell+1}+\ell P_{\ell-1}}{\ell+1}
\end{aligned}
$$

$1-\mu^2=\frac{2}{3}(P_0-P_2)$

$$
\begin{aligned}
& \int _{-1}^{1}d\mu P_0(\mu)=2 \
& \int _{-1}^{1}d\mu P_1(\mu)=0 \
& \int _{-1}^{1}d\mu P_2(\mu)=0 \
& \int _{-1}^{1}d\mu P_3(\mu)=0 \
& \int _{-1}^{1}d\mu P_\ell(\mu)=0,\ell>0 \text{\quad to be verified}\
\end{aligned}
$$

$$
\int_{-1}^{1}d\mu P_\ell P_{\ell’}=\delta_{\ell \ell’}\frac{2}{2\ell+1}
$$

Boltzmann hierarchy

Temperature

$$
\left{
\begin{aligned}
& \Theta_\ell \equiv \frac{1}{(-i)^\ell} \int_{-1}^1 \frac{d \mu}{2} P_\ell(\mu) \Theta \

&\Theta^{\prime}+ik\mu \Theta=-\Phi^{\prime}-i k\mu\Psi-\tau^{\prime}\left[\Theta_0-\Theta+\mu u_b-\frac{1}{2} P_{2}(\mu) \Pi\right] \
&\Pi=\Theta_2+\Theta_{p, 2}+\Theta_{p, 0}
\end{aligned}
\right.
$$

$$
\left{
\begin{aligned}
&\Theta_0’+k\Theta_1=-\Phi’\
&\Theta_1’+\frac{2}{3}k\Theta_2-\frac{1}{3}k\Theta_0=\frac{1}{3}k\Psi+\tau’\left[\Theta_1-\frac{i}{3}u_b\right]\
&\Theta_2’+\frac{3}{5}k\Theta_3-\frac{2}{5}k\Theta_1=-\tau’\left[-\Theta_2+\frac{\Pi}{10}\right]=\tau’\left(\frac{9}{10}\Theta_2-\frac{1}{10}\Theta_{p,2}-\frac{1}{10}\Theta_{p,0}
\right)\
&\Theta_\ell’+ik\frac{1}{(-i)^\ell}\int^1_{-1}\frac{d\mu}{2}\mu P_\ell\Theta=\tau’\Theta_\ell,\ell>2
\end{aligned}
\right.
$$

Polarization

$$
\left{
\begin{aligned}
& \Theta_\ell \equiv \frac{1}{(-i)^\ell} \int_{-1}^1 \frac{d \mu}{2} P_\ell(\mu) \Theta \

&\Theta^{\prime}p+ik\mu \Theta_p=-\tau^{\prime}\left[-\Theta_p+\frac{1}{2} (P_0-P{2}) \Pi\right] \
&\Pi=\Theta_2+\Theta_{p, 2}+\Theta_{p, 0}
\end{aligned}
\right.
$$

$$
\left{
\begin{aligned}
&\Theta_{p0}’+k\Theta_{p1}=-\tau’(\frac{1}{2}\Theta_{p0}-\frac{1}{2}\Theta_{p2}-\frac{1}{2}\Theta_2)\
&\Theta_{p1}’+\frac{2}{3}k\Theta_{p2}-\frac{1}{3}k\Theta_{p0}=\tau’\Theta_{p1}\
&\Theta_{p2}’+\frac{3}{5}k\Theta_{p3}-\frac{2}{5}k\Theta_{p1}=\tau’\left[\Theta_2-\frac{\Pi}{10}\right]=\tau’\left(\frac{9}{10}\Theta_{p2}-\frac{1}{10}\Theta_{2}-\frac{1}{10}\Theta_{p,0}
\right)\
&\Theta_{p\ell}’+ik\frac{1}{(-i)^\ell}\int^1_{-1}\frac{d\mu}{2}\mu P_\ell\Theta_p=\tau’\Theta_{p\ell},\ell>2
\end{aligned}
\right.
$$
